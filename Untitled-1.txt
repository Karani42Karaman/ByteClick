kSmoothingInput  = input.int(3, title="K Smoothing")
dSmoothingInput  = input.int(3, title="D Smoothing")

overboughtLevel  = input.int(80, title="Overbought Level")
oversoldLevel    = input.int(20, title="Oversold Level")

extremeHigh      = input.int(82, title="Extreme High Filter")
extremeLow       = input.int(18, title="Extreme Low Filter")

emaLength        = input.int(100, title="Trend EMA Length")

// === STOCHASTIC ===
k = ta.sma(ta.stoch(close, high, low, lengthInput), kSmoothingInput)
d = ta.sma(k, dSmoothingInput)

// === TREND FILTER ===
emaTrend = ta.ema(close, emaLength)

// === EXTRA FILTERS ===
// RSI
rsi = ta.rsi(close, 14)

// Volume
volMA = ta.sma(volume, 20)
volumeFilter = volume > volMA

// Candle filter
bullCandle = close > open
bearCandle = close < open

// === SIGNALS ===
longSignal = ta.crossover(d, oversoldLevel) and
     d[1] < extremeLow and
     close > emaTrend and
     rsi > 50 and
     volumeFilter and
     bullCandle and
     barstate.isconfirmed

shortSignal = ta.crossunder(d, overboughtLevel) and
     d[1] > extremeHigh and
     close < emaTrend and
     rsi < 50 and
     volumeFilter and
     bearCandle and
     barstate.isconfirmed

// === PLOT ===
plot(emaTrend, title="EMA Trend", color=color.orange)

plotshape(longSignal,
     title="LONG",
     location=location.belowbar,
     color=color.green,
     style=shape.labelup,
     text="LONG",
     size=size.small)

plotshape(shortSignal,
     title="SHORT",
